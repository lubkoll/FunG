36 template <class M, class Function>
37 void runTest(Function& f)
38 {
39  using std::cout;
40  using std::endl;
41  auto nUpdates = 10u * 1000u * 1000u;
42  auto nTimes = 1;
43  auto nEval = nUpdates * nTimes;
44 
45  M a;
46  a.eye();
47  M da0 = 2*a, da1 = 3*a, da2 = 4*a;
48  auto val = 0.;
49 
50  cout << "number of updates: " << nUpdates << endl;
51  boost::timer::cpu_timer timer;
52  for(auto i=0u; i<nUpdates; ++i) f.update(a);
53  cout << "updates: " << boost::timer::format(timer.elapsed());
54  timer.stop();
55  cout << "number of evaluations: " << nEval << endl;
56  timer.start();
57  for(auto i=0u; i<nEval; ++i) val = f();
58  cout << "d0: " << boost::timer::format(timer.elapsed());
59  cout << "value: " << val << endl;
60  timer.stop();
61  timer.start();
62  for(auto i=0u; i<nEval; ++i) val = f.template d1<0>(da0);
63  cout << "d1: " << boost::timer::format(timer.elapsed());
64  cout << "value: " << val << endl;
65  timer.stop();
66  timer.start();
67  for(auto i=0u; i<nEval; ++i) val = f.template d2<0,0>(da0,da1);
68  cout << "d2: " << boost::timer::format(timer.elapsed());
69  cout << "value: " << val << endl;
70  timer.stop();
71  timer.start();
72  for(auto i=0u; i<nEval; ++i) val = f.template d3<0,0,0>(da0,da1,da2);
73  cout << "d3: " << boost::timer::format(timer.elapsed());
74  cout << "value: " << val << endl;
75  cout << endl;
76 }

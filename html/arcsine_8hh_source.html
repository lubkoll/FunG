54  void update(double x)

56 #ifdef FUNG_ENABLE_EXCEPTIONS
57  if( x < -1 || x > 1 ) throw OutOfDomainException("ASin","[-1,1]",x,__FILE__,__LINE__);

59  value = ::asin(x);
